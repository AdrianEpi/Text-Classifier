204 void calculateError (int& argc, char* argv[]) {
205  if (argc != 4) {
206  std::cout << std::endl << "Error, the program needs 4 arguments to calculate the success and error percentage:" << std::endl << "\t bin/textClassifier -e classifiedResume expectedResume" << std::endl;
207  exit(1);
208  }
209  std::string expected = argv[3];
210  std::string received = argv[2];
211  std::vector<std::string> expect;
212  std::vector<std::string> receive;
213  std::string tmp = "";
214  std::ifstream file(expected, std::ios::in);
215  if (file.fail()) {
216  std::cout << std::endl << "Error 404," << expected << " file not found." << std::endl;
217  exit(1);
218  }
219  else {
220  while (!file.eof()) {
221  file >> tmp;
222  expect.push_back(tmp);
223  }
224  file.close();
225  }
226  std::ifstream file2(received, std::ios::in);
227  if (file2.fail()) {
228  std::cout << std::endl << "Error 404," << received << " file not found." << std::endl;
229  exit(1);
230  }
231  else {
232  file2 >> tmp;
233  while (!file2.eof()) {
234  file2 >> tmp;
235  receive.push_back(tmp);
236  }
237  file2.close();
238  }
239 
240  int size = 0;
241  if (expect.size() < receive.size()) {
242  size = expect.size();
243  }
244  else {
245  size = receive.size();
246  }
247  int correct = 0;
248  for (int i = 0; i < size; i++) {
249  if (expect[i] == receive[i]) {
250  correct++;
251  }
252  }
253  float percentage = correct;
254  percentage /= size;
255  percentage *= 100;
256  std::cout << std::endl << "Success percentage: " << correct << " / " << size << " = " << percentage << " %.";
257  std::cout << std::endl << "Error percentage: " << (size - correct) << " / " << size << " = " << (100 - percentage) << " %." << std::endl;
258 }
